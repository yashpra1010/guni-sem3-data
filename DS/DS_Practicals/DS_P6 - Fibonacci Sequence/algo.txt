void TowerOfHanoi(int n,char A,char B,char C)
{
	if(n>0) 
	{
		TowerOfHanoi(n-1,A,C,B);
		printf("\nTop disk moved from %c to %c",A,B);
		TowerOfHanoi(n-1,C,B,A);
	}
}
