#include<stdio.h>
main()
{	
	struct Batch
	{
		int Entry_Year;
		int Exit_Year;
	}
	struct Student
	{
		int Enroll_No;
		char Name[10];
		char Branch[10];
		struct Batch Batch1;
	};
	struct Student Student1;
 
	//Student1->Batch1->Entry_Year/Exit_Year	
	printf("Enter Your Enrollment Number:\n");
	scanf("%d",&Student1.Enroll_No);
	printf("Enter Your Name:\n");
	scanf("%s",Student1.Name);
	printf("Enter Your Branch:\n");
	scanf("%s",Student1.Branch);
	printf("Enter Batch (Starting Year):\n");
	scanf("%d",&Student1.Batch1.Entry_Year);
	printf("Enter Batch (Ending Year):\n");
	scanf("%d",&Student1.Batch1.Exit_Year);

	printf("Student Details");
	printf("Enrollment Number: %d\nName: %s\nBranch: %s\n",Student1.Enroll_No,Student1.Name,Student1.Branch);
	printf("Batch: %d-%d\n",Student1.Batch1.Entry_Year,Student1.Batch1.Exit_Year);
}
